#include <limits.h>
#include <float.h>

    printf("char max:%d\n",CHAR_MAX);
    printf("char min:%d\n",CHAR_MIN);

    printf("int min:%d\n",INT_MIN);

    printf("long min:%ld\n",LONG_MIN);
