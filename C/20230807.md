void merge(int* nums1, int nums1Size, int m, int* nums2, int nums2Size, int n){

    int* tmp[m+n];

    int i,j,k;
    i = 0, j = 0, k = 0;

    while (i<m && j<n){

        if (nums1[i] < nums2[j]){
            tmp[k] = nums1[i];
            i++;
        }else{
            tmp[k] = nums2[j];
            j++;
        }
        k++;
    }

    while (i<m){
        tmp[k] = nums1[i];
        i++;
        k++;
    }

    while (j<n){
        tmp[k] = nums2[j];
        j++;
        k++;
    }

    for(int i=0;i<m+n;i++){
        nums1[i] = tmp[i];
    }

}
