```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        
        length = len(nums)
        
        for i in range(length - 1):
            diff = target - nums[i]
            
            if diff in nums[i+1:]:
                return [i, nums[i+1:].index(diff)+ i +1]
```

##### 0009 Palindrome Number

```python
# 2022.08.28
class Solution:
    def isPalindrome(self, x: int) -> bool:
        result = True
        
        if x < 0:
            result = False
        
        else:
            x = str(x)
            length = len(str(x))
            n = length //2
            
            for i in range(n):
                
                if x[i] != x[length -1 - i]:
                    result = False
                    break

        return result
```

##### 0013 Roman to Integer
```python
# 2022.08.28
class Solution:
    def romanToInt(self, s: str) -> int:
        
        numerals = {'I':1, 'V':5, 'X':10, 'L':50, 'C':100, 'D':500, 'M': 1000} 
        sum = 0

        for i in range(len(s)-1):
            if numerals[s[i]] >= numerals[s[i+1]]:
                sum += numerals[s[i]]
            else:
                sum -= numerals[s[i]]
        
        sum += numerals[s[len(s)-1]]
        
        return sum
```
##### 0014 Longest Common Prefix

```python
# 2022.08.28
class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        
        length =[]
        output = ""
        
        if output in strs:
            return output
        
        for i in strs:
            length.append(len(i))
            
        for j in range(min(length)):
            for k in strs:
                if strs[0][j] != k[j]:
                    return output
                
            output += strs[0][j]
        
        return output
```

##### 0020 Valid Parentheses

```python
# 2022.08.28
class Solution:
    def isValid(self, s: str) -> bool:
        
        if len(s) %2 != 0:
            return False
        
        tmp = []
        for i in range(len(s)):
            
            if s[i] == '(' or s[i] == '[' or s[i] == '{':
                tmp.append(s[i])
            elif tmp == [] and (s[i] == ')' or s[i] == ']' or s[i] == '}'):
                return False
            elif s[i] == ')' and tmp[-1] == '(':
                tmp.pop()
            elif s[i] == ']' and tmp[-1] == '[':
                tmp.pop()
            elif s[i] == '}' and tmp[-1] == '{':
                tmp.pop()
            else:
                return False
                     
        if tmp == []:
            return True
        else:
            return False
```

##### 0021 Merge Two Sorted Lists
```python
# 2022.08.28
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:
        
        dummy = cur = ListNode(None)
        
        while list1 and list2:
            
            if list1.val <= list2.val:
                cur.next = list1
                list1 = list1.next
            else:
                cur.next = list2
                list2 = list2.next
            cur = cur.next
            
        if list1 == None:
            cur.next = list2
        if list2 == None:
            cur.next = list1
            
        return dummy.next
```

##### 0026 Remove Duplicates from Sorted Array
```python
class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        
        k = 0
        
        for i in range(len(nums)):
            if i == 0 or nums[i] != nums[i-1]:
                nums[k] = nums[i]
                k += 1
        return k
```
